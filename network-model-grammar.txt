<network-model> ::= network <identifier> = assets : <asset-list> facts : <fact-list> .
<asset-list>    ::= <identifier> <asset-list> ; | <EOL>
<fact-list>     ::= <fact> <fact-list> | <EOL>
<fact>          ::= quality : <identifier> , <statement> ;
                  | topology : <identifier> <direction> <identifier> , <statement> ;

<statement>     ::= <identifier> <eq> <value>
<identifier>    ::= [A-Za-z][A-Za-z0-9_]*

<value>         ::= <string>
                  | <number>
<string>        ::= [A-Za-z0-9_]*
<number>        ::= [1-9][0-9]*

<operator>      ::= <eqlop>
                  | <relop>
<eqlop>         ::= =
                  | :=
<relop>         ::= <
                  | >
                  | <=
                  | >=

<direction>     ::= ->
                  | <->

<whitespace>    ::= " "
                  | \t
                  | \n
